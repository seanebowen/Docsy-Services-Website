function fmt12(h){return(h%12||12)+':00'+(h>=12?' PM':' AM');}
function fmtTime(d){return d.toLocaleTimeString([],{hour:'numeric',minute:'2-digit'});}
function fmtDay(d){
  var now=new Date(),tom=new Date(now);tom.setDate(tom.getDate()+1);
  if(d.toDateString()===now.toDateString())return'Today '+fmtTime(d);
  if(d.toDateString()===tom.toDateString())return'Tomorrow '+fmtTime(d);
  return d.toLocaleDateString([],{weekday:'short',month:'short',day:'numeric'})+' '+fmtTime(d);
}
function isOpen(){var h=new Date().getHours();return h>=OPEN_H&&h<CLOSE_H;}
function getCopy(div,state){
  var c=Object.assign({},div[state]);
  if(c.h)c.h=c.h.replace('{OPEN}',fmt12(OPEN_H));
  if(c.s)c.s=c.s.replace('{OPEN}',fmt12(OPEN_H));
  return c;
}

/* ---- PAGE SWITCHER ---- */
function switchPage(p){
  ['home','division','copy','setup','embed'].forEach(function(id){document.getElementById('page-'+id).classList.add('hidden');});
  document.getElementById('page-'+p).classList.remove('hidden');
  document.querySelectorAll('.ps-btn').forEach(function(b,i){
    b.classList.remove('active');
    if(['home','division','copy','setup','embed'][i]===p)b.classList.add('active');
  });
  if(p==='copy')buildCopyTab();
}

function copyCode(id,btn){
  navigator.clipboard.writeText(document.getElementById(id).textContent).then(function(){
    var orig=btn.textContent;btn.textContent='Copied!';
    setTimeout(function(){btn.textContent=orig;},2000);
  });
}
